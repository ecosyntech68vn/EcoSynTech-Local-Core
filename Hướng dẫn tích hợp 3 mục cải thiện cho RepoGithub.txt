Bước 1.2: Thêm Service Worker (public/sw.js)
javascript
const CACHE_NAME = 'ecosyntech-v1';
const STATIC_ASSETS = [
  '/',
  '/index.html',
  '/css/app.css',
  '/js/app.js',
  '/manifest.json',
  '/icons/icon-192x192.png'
  // Thêm các file CSS/JS quan trọng khác
];

// Install: cache tài nguyên tĩnh
self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME).then(cache => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

// Activate: xóa cache cũ
self.addEventListener('activate', event => {
  event.waitUntil(
    caches.keys().then(keys => Promise.all(
      keys.filter(key => key !== CACHE_NAME).map(key => caches.delete(key))
    ))
  );
  self.clients.claim();
});

// Fetch: stale-while-revalidate cho API, cache-first cho tài nguyên
self.addEventListener('fetch', event => {
  const { request } = event;
  const url = new URL(request.url);

  // API calls: ưu tiên mạng, fallback cache
  if (url.pathname.startsWith('/api/')) {
    event.respondWith(
      fetch(request)
        .then(response => {
          const responseCopy = response.clone();
          caches.open(CACHE_NAME).then(cache => cache.put(request, responseCopy));
          return response;
        })
        .catch(() => caches.match(request))
    );
    return;
  }

  // Tài nguyên tĩnh: cache first
  event.respondWith(
    caches.match(request).then(cached => cached || fetch(request))
  );
});

2. Hoàn thiện cảnh báo Zalo & giọng nói
Bước 2.1: Tích hợp Zalo Official Account API
Giả sử bạn đã có access_token và oa_id. Tạo module services/zaloService.js:

javascript
const axios = require('axios');

class ZaloService {
  constructor(accessToken, oaId) {
    this.accessToken = accessToken;
    this.oaId = oaId;
    this.apiUrl = 'https://openapi.zalo.me/v2.0/oa/message';
  }

  async sendText(userId, message) {
    try {
      const response = await axios.post(this.apiUrl, {
        recipient: { user_id: userId },
        message: { text: message }
      }, {
        headers: {
          'access_token': this.accessToken,
          'Content-Type': 'application/json'
        }
      });
      return response.data;
    } catch (error) {
      console.error('Zalo send error:', error.response?.data || error.message);
      return null;
    }
  }

  // Gửi cảnh báo giọng nói (Zalo hỗ trợ gửi file âm thanh)
  async sendVoice(userId, voiceUrl) {
    try {
      const response = await axios.post(this.apiUrl, {
        recipient: { user_id: userId },
        message: {
          type: 'voice',
          voice: { url: voiceUrl, duration: 15 } // URL file .mp3
        }
      }, {
        headers: { 'access_token': this.accessToken }
      });
      return response.data;
    } catch (error) {
      console.error('Zalo voice error:', error);
      return null;
    }
  }
}

module.exports = ZaloService;
Bước 2.2: Tạo hàm chuyển văn bản thành giọng nói (sử dụng Google Text-to-Speech miễn phí hoặc VietTTS)
Cài đặt gói google-tts-api:

bash
npm install google-tts-api
Tạo services/ttsService.js:

javascript
const gtts = require('google-tts-api');
const fs = require('fs');
const path = require('path');

async function textToSpeech(text, lang = 'vi') {
  const url = gtts.getAudioUrl(text, { lang, slow: false });
  // Tải file âm thanh tạm thời
  const response = await fetch(url);
  const buffer = await response.arrayBuffer();
  const tempPath = path.join(__dirname, '../temp', `alert_${Date.now()}.mp3`);
  fs.writeFileSync(tempPath, Buffer.from(buffer));
  return tempPath;
}

module.exports = { textToSpeech };
Bước 2.3: Tích hợp vào luồng cảnh báo hiện có (ví dụ trong skills/monitoring-alert.js)
Mở file skills/monitoring-alert.js (hoặc nơi xử lý cảnh báo). Thêm code sau:

javascript
const ZaloService = require('../services/zaloService');
const { textToSpeech } = require('../services/ttsService');

const zalo = new ZaloService(process.env.ZALO_ACCESS_TOKEN, process.env.ZALO_OA_ID);

async function sendAlert(deviceId, message, userIds) {
  // Gửi Telegram (đã có)
  // Gửi Zalo text
  for (const userId of userIds) {
    await zalo.sendText(userId, message);
    // Nếu cảnh báo nghiêm trọng, gửi thêm giọng nói
    if (message.includes('khẩn') || message.includes('báo động')) {
      const voicePath = await textToSpeech(message);
      // Upload voicePath lên cloud (hoặc serve từ local) và gửi URL
      const voiceUrl = `https://your-domain.com/temp/${path.basename(voicePath)}`;
      await zalo.sendVoice(userId, voiceUrl);
    }
  }
}

// routes/sensor-data.js

Dữ liệu từ ESP32 vẫn được lưu vào SQLite bình thường, PWA và chế độ đơn giản chỉ đọc từ API có sẵn.

Tổng kết các bước tích hợp
Mục	Việc cần làm
PWA	Tạo manifest.json, sw.js, đăng ký SW, thêm link manifest vào HTML, tạo icon.
Zalo + Voice	Cài đặt axios, google-tts-api; tạo zaloService.js, ttsService.js; tích hợp vào monitoring-alert; thêm biến môi trường.
Chế độ đơn giản	Tạo simple.html, sửa app.js để chuyển chế độ, đảm bảo API /api/sensor-data/latest hoạt động.
Sau khi thực hiện, hệ thống sẽ có khả năng offline-first, cảnh báo giọng nói qua Zalo, và giao diện siêu đơn giản cho người lớn tuổi. Tất cả đều đồng bộ hoàn toàn với dữ liệu và logic hiện tại.
